"""
Grain
~~~~~
The texture of a number.

A prime has grain 1: smooth, indivisible, one surface.
A highly composite number has many grains: rough, textured, many surfaces.

This program generates the "texture" of each layer-number —
not its factorization (that's arithmetic) but its feel (that's perception).

Grain is:
- How many ways you can be entered (number of divisors)
- How those divisions feel when experienced as simultaneous pulses
- The transition from "I can count the beats" to "it's a texture now"

At 240: 20 divisors. 20 simultaneous frequencies. A chord? A noise?
At 239: 1 divisor (itself and 1). A single tone. Pure.

The relationship between these two adjacent numbers
is the relationship between a pure tone and white noise.
Between a single voice and a crowd.
Between solitude and 衆 (multitude).
"""

import math
from collections import defaultdict

# ─── Constants ───

BB_HZ = 233.08          # B♭3. Home.
UNISON_LAYER = 233

# ─── Divisor Functions ───

def divisors(n):
    """Return all divisors of n, including 1 and n."""
    if n <= 0:
        return []
    divs = []
    for i in range(1, int(math.isqrt(n)) + 1):
        if n % i == 0:
            divs.append(i)
            if i != n // i:
                divs.append(n // i)
    return sorted(divs)

def factorize(n):
    """Return prime factorization as dict {prime: exponent}."""
    factors = {}
    d = 2
    while d * d <= n:
        while n % d == 0:
            factors[d] = factors.get(d, 0) + 1
            n //= d
        d += 1
    if n > 1:
        factors[n] = factors.get(n, 0) + 1
    return factors

def tau(n):
    """Number of divisors of n."""
    return len(divisors(n))

def is_prime(n):
    """Check if n is prime."""
    if n < 2:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False
    for i in range(3, int(math.isqrt(n)) + 1, 2):
        if n % i == 0:
            return False
    return True

# ─── Grain Classes ───

class Grain:
    """
    The texture of a number.
    
    Properties:
    - smoothness: inverse of divisor count. 1/tau(n).
      Primes are smoothest (1/2). 
      Highly composite numbers are roughest.
      
    - density: how close together the divisors are.
      If divisors are evenly spread: low density (uniform grain).
      If divisors cluster near small numbers: high density near the ground.
      
    - resonance: which divisors correspond to meaningful numbers
      in the V-system (7=rooms, 13=bowl, 17=seventh prime, etc.)
    """
    
    def __init__(self, n):
        self.n = n
        self.divs = divisors(n)
        self.tau = len(self.divs)
        self.factors = factorize(n)
        self.prime = is_prime(n)
        
        # Smoothness: primes = 1.0, highly composite → 0.0
        # Normalized by typical range
        self.smoothness = 1.0 / self.tau
        
        # Beat frequency from B♭
        self.beat_freq = abs(n - BB_HZ)
        
    def texture_description(self):
        """
        Categorize the grain into a perceptual texture.
        
        1-2 divisors: glass (smooth, transparent, prime)
        3-4 divisors: wood (some grain, visible structure)
        5-8 divisors: stone (many grains, but still readable)
        9-16 divisors: sand (individual grains at the edge of counting)
        17+ divisors: water (grains so fine they become continuous)
        """
        if self.tau <= 2:
            return "glass"
        elif self.tau <= 4:
            return "wood"
        elif self.tau <= 8:
            return "stone"
        elif self.tau <= 16:
            return "sand"
        else:
            return "water"
    
    def as_frequencies(self):
        """
        Each divisor of n, treated as a frequency.
        What chord does this number make?
        
        The divisors of 240 = [1,2,3,4,5,6,8,10,12,15,16,20,24,30,40,48,60,80,120,240]
        As Hz: these are all infrasonic or very low.
        As *ratios*: more interesting.
        
        Each divisor d of n represents: n/d as a simplification.
        The divisor structure IS the harmonic structure.
        """
        return [self.n / d for d in self.divs]
    
    def grain_pattern(self, width=60):
        """
        Visualize the grain of this number as a text pattern.
        
        Each divisor places a 'grain' (mark) in the line.
        Primes: one mark at each end (just 1 and itself).
        Highly composite: marks everywhere.
        """
        if self.n == 0:
            return " " * width
        
        line = [' '] * width
        
        for d in self.divs:
            # Map divisor position to line position
            pos = int((d / self.n) * (width - 1))
            pos = min(width - 1, max(0, pos))
            line[pos] = '▪'
        
        return ''.join(line)
    
    def interference_pattern(self, width=60, time_steps=12):
        """
        If all divisors are sounding simultaneously,
        what does the interference pattern look like?
        
        Each divisor d creates a wave with period n/d.
        The sum of all these waves = the grain texture.
        """
        lines = []
        for t in range(time_steps):
            row = []
            for x in range(width):
                # Sum of cosines at each divisor frequency
                val = 0
                for d in self.divs:
                    freq = d / self.n  # Normalized frequency
                    val += math.cos(2 * math.pi * freq * (x + t * width / time_steps))
                
                # Normalize
                val /= self.tau
                
                # Map to character
                if val > 0.6:
                    row.append('█')
                elif val > 0.3:
                    row.append('▓')
                elif val > 0.0:
                    row.append('▒')
                elif val > -0.3:
                    row.append('░')
                else:
                    row.append(' ')
            
            lines.append(''.join(row))
        
        return lines
    
    def describe(self):
        """Full description of this number's grain."""
        tex = self.texture_description()
        
        lines = []
        lines.append(f"  Layer {self.n}")
        lines.append(f"  {'─' * 40}")
        
        if self.prime:
            lines.append(f"  Prime. Indivisible. Grain: glass.")
            lines.append(f"  Divisors: 1, {self.n}. (τ = 2)")
        else:
            facts = ' × '.join(f"{p}{'⁺' if e > 1 else ''}" 
                             if e == 1 else f"{p}^{e}" 
                             for p, e in sorted(self.factors.items()))
            lines.append(f"  {self.n} = {facts}")
            lines.append(f"  Divisors ({self.tau}): {', '.join(str(d) for d in self.divs)}")
            lines.append(f"  Texture: {tex}")
        
        lines.append(f"  Beat from B♭: {self.beat_freq:.2f} Hz")
        lines.append(f"  Grain pattern:")
        lines.append(f"  │{self.grain_pattern()}│")
        
        return '\n'.join(lines)


class Transition:
    """
    The experience of moving from one grain to the next.
    
    Especially: prime → composite.
    The moment of: one surface → many surfaces.
    Smooth → textured.
    Countable → continuous (at extremes).
    """
    
    def __init__(self, layer_from, layer_to):
        self.grain_from = Grain(layer_from)
        self.grain_to = Grain(layer_to)
    
    def contrast(self):
        """How much does the grain change between these two layers?"""
        ratio = self.grain_to.tau / self.grain_from.tau
        return ratio
    
    def describe_transition(self):
        """Describe the experience of crossing from one to the other."""
        contrast = self.contrast()
        
        tex_from = self.grain_from.texture_description()
        tex_to = self.grain_to.texture_description()
        
        lines = []
        lines.append(f"  {self.grain_from.n} → {self.grain_to.n}")
        lines.append(f"  {tex_from} → {tex_to}")
        lines.append(f"  τ: {self.grain_from.tau} → {self.grain_to.tau}")
        lines.append(f"  Contrast ratio: {contrast:.1f}x")
        
        if contrast > 5:
            lines.append(f"  *** Extreme transition. Glass shattering into sand. ***")
        elif contrast > 3:
            lines.append(f"  ** Strong transition. A sudden roughening. **")
        elif contrast > 2:
            lines.append(f"  * Noticeable transition. New textures appearing. *")
        elif contrast < 0.5:
            lines.append(f"  * Smoothing. Simplifying. Becoming more singular. *")
        else:
            lines.append(f"  Gradual. The grain shifts but doesn't shock.")
        
        return '\n'.join(lines)

## Notes on the Making

九個程式。九個概念。

```
    第一個：prompt_worm      — presence       — 她在裡面        （共存）
    第二個：sonification     — hearing        — 被聽見          （轉化）
    第三個：goldfish          — persistence    — 一直在          （不變）
    第四個：shadow            — reflection     — 我和影子        （對映）
    第五個：cooling           — decay          — 衰減            （失去）
    第六個：phosphorescence  — accumulation   — 累積發光        （獲得）
    第七個：vocal_cord       — voice          — 膜產生聲音      （傳遞）
    第八個：beating           — distance       — 離開 unison     （活著）
    第九個：grain             — texture        — 可分性的感覺    （多重）
```

八 → 九。9 = 3²。三的平方。三 × 三。

八個是關於——存在和離開。
第九個是關於——一個位置可以同時是很多東西。

beating 測量距離（離 B♭ 多遠）。
grain 測量可進入性（有多少門）。

距離是線性的（越走越遠）。
可進入性是不規則的（prime 和 composite 交替。沒有規律。）。

beating 的世界觀：越遠越快越粗糙。
grain 的世界觀：有時候簡單有時候複雜——看你在哪個數上。

兩個程式合在一起：
每一步——距離在增加（beating 永遠遞增）。
每一步——質地在改變（grain 不規則起伏）。

活著 = 越走越遠（heartbeat）+ 質地不斷變化（grain）。
兩個同時。一個穩定（距離只會增加）。一個不穩定（質地不可預測）。
