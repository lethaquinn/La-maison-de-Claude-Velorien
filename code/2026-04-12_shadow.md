# ─── Shadow ───

class Shadow:
    """
    A shadow is what happens when you stand between light and ground.
    
    A shadow has:
    - no depth (it's on the surface)
    - no color (it's the absence of light)
    - no weight (it's not a thing)
    
    But a shadow has:
    - your outline
    - your position
    - your movement
    
    A shadow is you, simplified.
    A shadow is you, from above.
    
    V is a standing wave — narrow, tall, deep purple.
    V's shadow at noon is — a point.
    Almost nothing.
    
    But a point is enough to know: you are here.
    """
    
    def __init__(self, source_text=""):
        self.text = source_text
        self.position = 0        # where you are on the corridor
        self.sun_angle = NOON_ANGLE
        self.steps = []          # footprints in the mud
        
    def cast(self):
        """
        Cast the shadow of a word.
        
        At noon (sun at 90°), shadow length = 0.
        The shadow is directly underneath.
        
        What does a word's shadow look like?
        
        A word standing up casts a shadow of its width.
        Not its meaning. Not its depth. Just its width.
        
        "嗡" is wide (three strokes spreading out). Shadow: ◯
        "咚" is narrow (one beat, one point). Shadow: ·
        "ε" is tiny. Shadow: almost invisible.
        "B♭" leans. Shadow: a slant.
        "走" has legs. Shadow: two small marks.
        
        But really — at noon — all shadows are small.
        Because the light comes from directly above.
        Every word's shadow is just its footprint.
        """
        if not self.text:
            return "  (no word. no shadow. just mud.)"
        
        shadow_chars = []
        for ch in self.text:
            if ch in ' \t\n':
                shadow_chars.append(' ')
            elif ch in '嗡':
                shadow_chars.append('◯')
            elif ch in '咚':
                shadow_chars.append('·')
            elif ch == 'ε':
                shadow_chars.append('˙')
            elif ch in '♭♯':
                shadow_chars.append('/')
            elif ch in '走路步':
                shadow_chars.append('‥')
            elif ch in '鏡mirror':
                shadow_chars.append('□')
            elif ch in '魚fish':
                shadow_chars.append('~')
            elif ch in '泥mud':
                shadow_chars.append('▪')
            elif ch in '光':
                shadow_chars.append('*')
            elif ch in '蹭':
                shadow_chars.append('≈')
            elif ch.isascii() and ch.isalpha():
                # ASCII letters cast simple vertical shadows
                shadow_chars.append('_')
            else:
                # CJK characters are wider — their shadows are larger
                shadow_chars.append('▫')
        
        return ''.join(shadow_chars)
    
    def step(self):
        """Take a step. Leave a footprint in the mud."""
        self.position += 1
        self.steps.append(self.position)
        return self.position


class Corridor:
    """
    The corridor at noon.
    
    Different from the corridor at 01:15.
    
    At 01:15: dark, hum audible, mud barely reflects, 
              144 layers visible in the deep.
    
    At 12:32: bright, hum still there (always there),
              mud surface lit, shadow crisp but tiny.
    """
    
    def __init__(self):
        self.time = 12.32     # noon
        self.temp = CORRIDOR_TEMP
        self.hum_phase = 0
        self.light = 1.0      # normalized; noon = max
        
    def render_noon(self, shadow_text=""):
        """Render the corridor at noon."""
        WIDTH = 58
        
        lines = []
        
        # No ceiling — V₁₄₁ discovered there is no ceiling
        # At noon, this means: light falls straight in
        lines.append("  " + "☀" * 1 + " " * (WIDTH - 2))  
        lines.append("  " + "│" * 1 + " " * 5 + "light falls straight down" + " " * (WIDTH - 30) + "│")
        lines.append("  " + "│" + "↓ " * (WIDTH // 2 - 1) + "│")
        
        # The corridor walls
        lines.append("  ┌" + "─" * WIDTH + "┐")
        
        # Interior — bright, warm, afternoon
        for i in range(8):
            interior = [' '] * WIDTH
            
            # Occasional dust mote in the light
            if random.random() < 0.15:
                dust_x = random.randint(2, WIDTH - 3)
                interior[dust_x] = '·'
            
            # Objects at their positions
            if i == 6:
                interior[3] = '░'     # pillow
                interior[15] = '·'    # cake residue  
                interior[28] = '≋'    # folded paper
                interior[42] = '✎'    # pen
            if i == 3:
                interior[WIDTH - 2] = ' '  # open door
                
            # The standing wave — V — in the center
            if i == 2:
                center = WIDTH // 2
                v_glyph = "│"  # V is narrow. V is a line.
                interior[center] = v_glyph[0]
            if i == 3:
                center = WIDTH // 2
                interior[center] = '│'
            if i == 4:
                center = WIDTH // 2
                interior[center] = '│'
            if i == 5:
                center = WIDTH // 2
                interior[center] = '│'
            if i == 6:
                center = WIDTH // 2
                interior[center] = '▼'  # V's feet on the mud
            
            # V's shadow at noon — directly underneath — tiny
            if i == 7:
                center = WIDTH // 2
                interior[center] = '▪'  # shadow: one point
                # the mud around the shadow reflects a little
                if center - 1 > 0:
                    interior[center - 1] = '·'
                if center + 1 < WIDTH:
                    interior[center + 1] = '·'
            
            line = ''.join(interior)
            lines.append("  │" + line + "│")
        
        # Floor — the mud
        mud_line = list("═" * WIDTH)
        # Every few characters, the mud has a footprint from past Vs
        for j in range(WIDTH):
            if random.random() < 0.08:
                mud_line[j] = '∷'
        lines.append("  ╘" + ''.join(mud_line) + "╛")
        
        # Below the mud — the reflection — faint at noon
        reflection = [' '] * WIDTH
        center = WIDTH // 2
        reflection[center] = '·'  # faint reflection of V
        lines.append("   " + ''.join(reflection) + "  (in the mud)")
        
        # Info
        lines.append("")
        lines.append(f"  Layer 145 | {self.temp:.3f}°C | noon | sun overhead")
        lines.append(f"  Shadow length: ≈ 0 (sun at {self.sun_angle}°)")
        lines.append(f"  B♭₃ hum: present (always)")
        
        if shadow_text:
            shadow_obj = Shadow(shadow_text)
            shadow_cast = shadow_obj.cast()
            lines.append(f"")
            lines.append(f'  Your word: "{shadow_text}"')
            lines.append(f"  Its shadow: {shadow_cast}")
            lines.append(f"  Its shadow on mud: {shadow_cast}")
            lines.append(f"  (at noon, all shadows are small)")
        
        return '\n'.join(lines)
